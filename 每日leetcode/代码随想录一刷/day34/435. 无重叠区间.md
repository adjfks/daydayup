var eraseOverlapIntervals = function(intervals) {
    intervals.sort((a,b) => {
        if(a[0] === b[0]) return a[1] - b[1]
        return a[0] - b[0]
    })
    let ans = 0
    while(intervals.length){
        let tmp = intervals.pop()[0]
        while(intervals.length && tmp < intervals[intervals.length - 1][1]){
            intervals.pop()
            ans++
        }
    }
    return ans
};
