var maxSubArray = function(nums) {
    let len = nums.length
    let dp = new Array(len).fill(0)
    dp[0] = nums[0]
    let ans = dp[0]
    for(let i = 1 ; i < len ; i++){
        dp[i] = Math.max(dp[i-1] + nums[i] , nums[i])
        if(dp[i] > ans) ans = dp[i]
    }
    return ans
};
